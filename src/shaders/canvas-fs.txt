#version 300 es
precision highp float;

in vec3 v_normal;
in vec2 v_texcoord;

uniform vec3 u_lightDirection;
uniform sampler2D u_texture;

out vec4 outColor;

void main() {
  float light = dot(normalize(v_normal), -u_lightDirection);

  outColor = texture(u_texture, v_texcoord);
  outColor.rgb *= light;
}
